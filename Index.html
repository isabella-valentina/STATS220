My meme

I decided to focus on my happy reaction when I realised that part of this project was creating our own meme. I chose this idea as it’s relevant and original, being based off of my own experience with this assignment.

# Project 1 Part B: Meme Creation

# First row: 
sad_man <- image_read("https://s3.amazonaws.com/cms.ipressroom.com/173/files/20205/5ef263842cfac20b533463c5_CHPR-Mental+Health+Services/CHPR-Mental+Health+Services_2870336a-2a3e-46a1-a440-b4eb36f5872d-prv.jpg") %>%
  image_crop("700x700")%>%
  image_scale("x500")

# caption
caption_one <- image_blank(width = 500, 
                          height = 500, 
                          color = "#000000") %>%
  image_annotate(text = "Thinking about my statistics project",
                 color = "#FFFFFF",
                 size = 27,
                 font = "Comic Sans",
                 gravity = "center")

# Second row: 
celebrating_man <- image_read("https://thumbs.dreamstime.com/b/man-using-laptop-building-online-business-under-money-rain-young-making-dollar-bills-cash-falling-down-beginner-entrepreneur-78648715.jpg")%>%
  image_crop("500x500", gravity = "center")


# caption
caption_two <- image_blank(width = 500, 
                       height = 500, 
                       color = "#000000") %>%
  image_annotate(text = "Realising I just have to make memes",
                 color = "#FFFFFF",
                 size = 27,
                 font = "Comic Sans",
                 gravity = "center")



# first using the approach we used above
man_vector <- c(sad_man, caption_one)
top_row <- image_append(man_vector)

happy_man_vector <- c(celebrating_man, caption_two)
bottom_row <- image_append(happy_man_vector)

meme <- c(top_row, bottom_row) %>%
  image_append(stack = TRUE) %>%
  image_scale(800)
print(meme)

# saving meme as PNG
image_write(meme, "final_meme.png")

My animated GIF

I wanted to create a GIF that showed a process, and immediately thought of a plant growing. I wanted to add to it to make it more original, so included a quote and made changes to saturation to create a more meaningful animation.

library(magick)

# reading images
pic_1 <- image_read("https://www.gardeningknowhow.com/wp-content/uploads/2020/07/seed-planting.jpg")
pic_2 <- image_read("https://www.gardeningknowhow.com/wp-content/uploads/2019/10/seedling.jpg")
pic_3 <- image_read("https://images.squarespace-cdn.com/content/v1/58cfd3b3ff7c501ac4387b35/1587674798974-93UQ4GUPDGSBS38NX95K/SoilBlog.jpg")
pic_4 <- image_read("https://www.gardeningknowhow.com/wp-content/uploads/2021/03/foliage.jpg") 

# creating vector count
counts <- 1 :4 %>% as.character()

# creating frames for gif
frame1 <- pic_1 %>%
  image_annotate("from", size = 500,color = "white", gravity = "center") %>%
  image_scale("200x200") %>%
  image_extent("500x500")%>%
  image_colorize(50,"green")%>%
  image_modulate(saturation = 25)
  
frame2 <- pic_2 %>%
  image_annotate("little things", size = 300,color = "white", gravity = "center")%>%
  image_scale("200x200") %>%
  image_extent("500x500")%>%
  image_colorize(60, color = "green")%>%
  image_modulate(brightness = 100, saturation = 50)
  
frame3 <- pic_3 %>%
    image_annotate("big things", size = 500,color = "white", gravity = "center")%>%
    image_scale("200x200") %>%
    image_extent("500x500")%>%
    image_colorize(60, color = "green")%>%
    image_modulate(brightness = 100, saturation = 75)
    
frame4 <- pic_4 %>%
    image_annotate("grow", size = 500,color = "white", gravity = "center")%>%
    image_scale("200x200") %>%
    image_extent("500x500")%>%
    image_colorize(60, color = "green")%>%
    image_modulate(brightness = 100, saturation = 105)

# final gif creation
frames <- c(frame1, frame2, frame3, frame4)
animation <- image_animate(frames, fps = 1)

# saving gif
image_write(animation, "grow.gif")
